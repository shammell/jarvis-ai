        self._init_client()

    def _init_client(self):
        api_key = os.getenv("GROQ_API_KEY")
        if api_key:
            try:
                from groq import Groq
                self.client = Groq(api_key=api_key)
            except ImportError:
                self.client = None
        else:
            self.client = None

    @classmethod
    def instance(cls):
        with cls._lock:
            if cls._instance is None:
                cls._instance = cls()
        return cls._instance

        if not self.client:
            raise LLMClientUnavailableError(
                "LLM client not initialized. Set GROQ_API_KEY in .env"
            )
        return self.client

import re

def _parse_prm_score(text: str, default: float = 0.5) -> float:
    """Robustly parse a 0.0-1.0 score from LLM text output."""
    # Try JSON first
    try:
        import json
        data = json.loads(text)
        val = float(data.get("score", default))
        return max(0.0, min(1.0, val))
    except Exception:
        pass
    # Fallback: regex extract first number
    m = re.search(r"([-+]?\d*\.\d+|\d+)", text)
    if not m:
        logger.warning("PRM returned no parseable score: %s", text[:100])
        return default
    try:
        val = float(m.group(0))
        return max(0.0, min(1.0, val))
    except Exception:
        return default

import yaml

def _parse_skill_file(self, skill_path: str) -> dict:
    with open(skill_path, 'r', encoding='utf-8') as f:
        content = f.read()
    
    metadata = {}
    body = content
    
    if content.startswith('---'):
        parts = content.split('---', 2)
        if len(parts) >= 3:
            try:
                metadata = yaml.safe_load(parts[1]) or {}
                body = parts[2]
            except yaml.YAMLError as e:
                logger.error("YAML parse error in skill %s: %s", skill_path, e)
                os.makedirs("logs", exist_ok=True)
                with open("logs/skill_loader_errors.log", "a") as le:
                    le.write(f"{skill_path}: {repr(e)}\n")
    
    return {"metadata": metadata, "body": body, "path": skill_path}
